def expand_status(status_list):

    stats = {
        "CURRENT": 0,
        "PLANNING": 0,
        "COMPLETED": 0,
        "PAUSED": 0,
        "DROPPED": 0
    }

    if status_list:
        for s in status_list:
            stats[s["status"]] = s["amount"]

    return pd.Series(stats)




url = "https://graphql.anilist.co"

query = """
query ($page: Int, $perPage: Int) {
  Page(page: $page, perPage: $perPage) {
     pageInfo {
        hasNextPage
    }
    media(type: MANGA) {
      id
      title { romaji english }
      chapters
      volumes
      averageScore
      meanScore
      genres
      popularity
      stats {
        statusDistribution {
          status
          amount
        }
      }
      description
      startDate { year }
    }
  }
}
"""

all_manga = []
page = 1

while True:
    variables = {
        "page": page,
        "perPage": 60  # max allowed
    }

    
    response = requests.post(url, json={"query": query, "variables": variables})
    data = response.json()


        
    page_data = data["data"]["Page"]["media"]
    all_manga.extend(page_data)
    
    if len(all_manga) >= 4000:
        break
        
    if not data["data"]["Page"]["pageInfo"]["hasNextPage"]:
        break

    page += 1
    time.sleep(5)
    
# now print ALL collected titles
for manga in all_manga:
    print(manga)




df = pd.json_normalize(all_manga)

status_df = df["stats.statusDistribution"].apply(expand_status)

df = pd.concat([df, status_df], axis=1)

df = df[df["stats.statusDistribution"].notnull()]

df = df[df["stats.statusDistribution"].str.len() > 0]

Conclusion¶

While many manga gain massive attention through high planning counts, our analysis shows that popularity does not guarantee completion. Genres like action and fantasy dominate engagement, but Comedy and Slice of Life titles tend to have higher drop-off rates. Interestingly, several highly rated manga still struggle with completion, suggesting that critical acclaim doesn’t always translate to sustained reader commitment.
